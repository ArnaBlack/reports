Анимации, привязанные к сколлу

animation-timeline: scroll(<scroller> <axis>)

scroller

  • nearest (default)
  • root
  • self

axis

  • block (default)
  • inline
  • x
  • y
scroll() 
          
          .container {
             overflow-y: scroll;
             scroll-timeline-name: --container-timeline;
          }

          .animated-box {
            animation-name: someAnimation;
            animation-timeline: --container-timeline;
          }

animation-timeline: view(<axis> || <view-timeline-inset>)

axis

  • block (default)
  • inline
  • x
  • y

view-timeline-inset

  • auto (default)
  • length-percentage {start, end?}
view() 
          
           .container {
             view-timeline-name: --container-view-timeline;
           }

          .animated-box {
            animation-name: someAnimation;
